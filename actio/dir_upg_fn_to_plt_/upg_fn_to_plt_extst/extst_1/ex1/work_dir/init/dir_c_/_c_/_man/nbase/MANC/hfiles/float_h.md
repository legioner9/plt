in TS 18661-3.  */

#if !__GNUC_PREREQ (7, 0) && __GLIBC_USE (IEC_60559_TYPES_EXT)

#  define FLT128_TRUE_MIN                                
__f128 (6.47517511943802511092443895822764655e-4966)
# endif
/* Types other than _Float128 are typedefs for other types with old
compilers.  */
